cd 'ссылка' - переход в папку
git clone ссылка - подключение к репрезиторию
git status - проверка файлов в папке
git add . - отслеживать файлы
git commit -m "комментарий" - отправить на локальный сервер
git push - отправить на GitHub
git branch - каналы
git branch ch1 - создать канал
git checkout ch1 - перейти на канал
